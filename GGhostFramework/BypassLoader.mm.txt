#import <Foundation/Foundation.h>
#import <dlfcn.h>
#import <mach-o/dyld.h>
#import <iostream>

// --- CONFIGURACIÓN ---
// Aquí debes poner el nombre que le diste a tu dylib camuflado
#define MOD_NAME "UnityAnalyticsCore.dylib" 
#define DELAY_SECONDS 8 

void cargarModFantasma() {
    NSLog(@"[GhostFramework] Iniciando fase de camuflaje...");
    
    // 1. Espera de seguridad (Evita detección al arranque)
    [NSThread sleepForTimeInterval:DELAY_SECONDS];
    
    // 2. Localizar la ruta del documento
    NSString *path = [[NSBundle mainBundle] bundlePath];
    NSString *dylibPath = [path stringByAppendingPathComponent:@MOD_NAME];
    
    // 3. Carga el dylib del mod de forma dinámica
    void *handle = dlopen([dylibPath UTF8String], RTLD_NOW);
    
    if (handle) {
        NSLog(@"[GhostFramework] Mod inyectado exitosamente bajo el radar.");
    } else {
        NSLog(@"[GhostFramework] Error: No se pudo cargar la librería de apoyo.");
    }
}

// Constructor: Se ejecuta automáticamente al abrir el juego
__attribute__((constructor))
static void initialize() {
    // Creamos un hilo separado para que el juego no se congele mientras esperamos
    dispatch_async(dispatch_get_global_queue(DISPATCH_QUEUE_PRIORITY_DEFAULT, 0), ^{
        cargarModFantasma();
    });
}
